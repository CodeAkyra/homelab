# PART 2 - VLANs, Layer-2 EhterChannel

###  DSW-A1, DSW-A2
  en
    conf t
      do sh cdp neig
      int range g1/0/4-5
        channel-group 1 mode desirable #cisco proprietary protocol PAgP
        exit
      do sh eth sum
      do wr

    # configure all links between access and distribution, including etherchannels, as trunk links.
    
      do sh cdp neig
      int range g1/0/1-3
        switchport mode trunk
        switchport nonegotiate
        switchport trunk native vlan 1000 #unused
        switchport trunk allowed vlan 10,20,40,99
        int p1
        switchport mode trunk
        switchport nonegotiate
        switchport trunk native vlan 1000
        switchport trunk allowed vlan 10,20,40,99
        exit
      do wr

      # create vtp domain and change the version to 2
      do sh vtp status
      vtp domain JeremysITLab
      vtp version 2
      do wr

      vlan 10
        name PCs
        vlan 20
        name Phones
        vlan 40
        name Wi-Fi
        vlan 99
        name Management
      
      # shut all unused ports
      int range g1/0/6-24
        shutdown
        int range g1/1/3-4
        shutdown
        exit
      do wr

### ASW-A1, ASW-A2, ASW-A3
  en
    conf t
      do sh cdp neig
      int range g0/1-2
        switchport mode trunk
        switchport nonegotiate
        switchport trunk native vlan 1000
        switchport trunk allowed vlan 10,20,40,99
        exit
      do wr

      # make sure that access switches join the domain and change the mode to client
      do sh vtp status
      vtp mode client
      do wr      
      
      # ASW-A1 only, remove hash
      # int f0/1
      #   switchport mode access
      #   switchport nonegotiate
      #   switchport access vlan 99
  
      # ASW-A2, ASW-A3 only
      # int f0/1
      #   switchport mode access
      #   switchport nonegotiate
      #   switchport access vlan 10
      #   switchport voice vlan 20
      #   do wr
      
      # shut unused ports
      # for ASW-A1
      int range f0/3-24
        shut
        exit
      # ASW-A2, ASW-A3
      int range f0/2-24
        shut
        exit
      do wr
